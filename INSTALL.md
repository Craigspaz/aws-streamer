# AWS Streamer - System Packages Installation

## Ubuntu

```
sudo apt-get update
sudo apt-get install \
    build-essential \
    cmake \
    pkg-config \
    python3-dev \
    python3-gi \
    gir1.2-gstreamer-1.0 \
    gir1.2-gst-plugins-base-1.0 \
    gstreamer1.0-tools \
    gstreamer1.0-plugins-base \
    gstreamer1.0-plugins-good \
    libgirepository1.0-dev \
    libgstreamer1.0-dev \
    libcairo2-dev \
    libpango1.0-dev \
    libjpeg-dev \
    libgif-dev \
    librsvg2-dev
```

## MacOS

```
brew install \
    autoconf \
    automake \
    coreutils \
    cmake \
    pkg-config \
    python@3.7 \
    pygobject3 \
    gstreamer \
    gst-plugins-base \
    gst-plugins-good \
    gst-plugins-ugly \
    gst-python \
    cairo \
    py3cairo
```
